document.getElementById('translateBtn').addEventListener('click', async () => { const inputText = document.getElementById('inputText').value; const sourceLang = document.getElementById('sourceLang').value; const targetLang = document.getElementById('targetLang').value; if (!inputText.trim()) { alert('Please enter text to translate.'); return; } try { const { text } = await translate(inputText, { from: sourceLang, to: targetLang }); document.getElementById('outputText').value = text; } catch (error) { console.error('Translation error:', error); alert('Translation failed. Please try again.'); } }); document.getElementById('swapBtn').addEventListener('click', () => { const source = document.getElementById('sourceLang'); const target = document.getElementById('targetLang'); const temp = source.value; source.value = target.value; target.value = temp; });
